4_SELECT CONCAT(p.apellido, ', ', p.nombre) AS paciente,

6_ SELECT i.numero_cama AS numero_cama, COUNT(i.numero_cama) AS usos
FROM ingresos i
GROUP BY i .numero_cama ORDER BY usos DESC;

7_SELECT 
CONCAT(p.apellido, ', ', p.nombre) AS paciente, COUNT(i.nro_ss) AS cantidad_ingresos
FROM pacientes p INNER JOIN ingresos i ON p.nro_ss = i.nro_ss
GROUP BY p.apellido, p.nombre
HAVING COUNT(i.nro_ss) >= 4
ORDER BY cantidad_ingresos DESC;
